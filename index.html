Наконец-то разобрался с декораторами! Держите простой пример для новичков 👇

def my_decorator(func):
    def wrapper():
        print("До функции")
        func()
        print("После функции")
    return wrapper

@my_decorator
def say_hello():
    print("Привет, мир!")
